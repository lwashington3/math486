\begin{problems}
	\problem The following data is modeled by a power law.
	\begin{equation}
		y = at^{c}
		\label{eq:1}
	\end{equation}
	\begin{table}[H]
		\centering
		\label{tab:1}
		\begin{tabular}{c|c}
			$t$ & $y$\\
			\hline
			1 & 2\\
			1 & 4\\
			2 & 5\\
			5 & 10\\
		\end{tabular}
	\end{table}
	\begin{problems}
		\subproblem Derive a linear model to replace~\eqref{eq:1}. \addanswer{Problem-1a}
		\subproblem Write a linear system $Ax=b$ with $x$ related to $a$ and $c$ using the data points. \addanswer{Problem-1b}
		\subproblem State the normal equations then solve, you can use a computer for the calculation. \addanswer{Problem-1c}
		\subproblem Plot the data points and the continuous function~\eqref{eq:1} using the computed values of $a$ and $c$ on the same graph. \addanswer{Problem-1d}
	\end{problems}

	\problem The table below gives car sales data starting at 1950 (year 0) in 5 year increments.
